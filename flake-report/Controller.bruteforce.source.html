4 def findallcombinations(actions_list):
5 """
6 This function will help to figure out all the possible combinations between all the actions
7 (each action can be used only once).
8 The function will also check that the spend doesn't go over 500€
9 """
10 final_combinations_list = []
11  
12 for i in range(len(actions_list)):
13 final_combinations_list.append([actions_list[i]]) # Every action alone is a combination
14  
15 # In the script below we test the different combinations
16 # within the limit of 500€ initial investment
17 for j in range(len(actions_list)):
18 counter = int(actions_list[i].value) # Will help to respect the max 500€ spend per customer
19 combination = [actions_list[i]]
20  
21 for k in range(j, len(actions_list)):
22 if counter + int(actions_list[k].value) <= max_spend_per_customer and k != i:
23 counter += int(actions_list[k].value)
24 combination.append(actions_list[k])
25 else: # If the ceiling of 500€ is reached or if we already used the action we skip
26 continue
27  
28 final_combinations_list.append(sorted(combination, key=lambda x: x.name))
29  
30 final_combinations_list_nodupl = []
31 # The loop below helps to remove the duplicated combinations
32 for item in final_combinations_list:
33 if item not in final_combinations_list_nodupl:
34 final_combinations_list_nodupl.append(item)
35 return final_combinations_list_nodupl

38 def combinationranking(combinations_list):
39 """
40 Out of a list of combinations of Action objects, the function will sort them depending on the
41 final benefit generated
42 :param combinations_list: list of Action objects
43 :return: sorted list of Action objects
44 """
45 combinations_ranking = []
46 for combination in combinations_list:
47 actions_list = []
48 combination_benefit = 0
49 initial_investment = 0
50 for action in combination:
51 actions_list.append(action.name)
52 combination_benefit += action.finalvalue()
53 initial_investment += float(action.value)
54 combinations_ranking.append((actions_list, initial_investment, combination_benefit))
55 combinations_ranking_final = sorted(combinations_ranking, key=lambda x: -x[2])
56 return combinations_ranking_final
